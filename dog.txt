local screenGui = Instance.new("ScreenGui")
local textLabel = Instance.new("TextLabel")

screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

textLabel.Size = UDim2.new(1, 0, 0, 50)  -- Full width, height of 50
textLabel.Position = UDim2.new(0.5, -100, 0, 10)  -- Centered at the top
textLabel.Text = "stellar.wave | loaded"
textLabel.TextScaled = true
textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)  -- White color
textLabel.BackgroundTransparency = 1
textLabel.Font = Enum.Font["Jura"]  -- Use the Jura font by its name
textLabel.Parent = screenGui

local function rainbowText()
    while true do
        for i = 0, 255 do
            local hue = i / 255
            local color = Color3.fromHSV(hue, 1, 1) -- Full saturation and brightness
            textLabel.TextColor3 = color
            wait(0.05) -- Adjust speed of color change
        end
    end
end

rainbowText()
