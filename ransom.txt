local replicatedStorage = game:GetService("ReplicatedStorage")
local debris = game:GetService("Debris")

-- Create a bomb model (you can customize this)
local bomb = Instance.new("Part")
bomb.Size = Vector3.new(1, 1, 1)
bomb.BrickColor = BrickColor.new("Bright red")
bomb.Material = Enum.Material.Neon
bomb.Anchored = false
bomb.CanCollide = true

local function spawnBomb()
    local newBomb = bomb:Clone()
    newBomb.Position = Vector3.new(math.random(-50, 50), 100, math.random(-50, 50)) -- Adjust the range as needed
    newBomb.Parent = game.Workspace

    -- Create an explosion when the bomb hits the ground
    newBomb.Touched:Connect(function(hit)
        if hit:IsA("BasePart") then
            local explosion = Instance.new("Explosion")
            explosion.Position = newBomb.Position
            explosion.BlastRadius = 10
            explosion.BlastPressure = 5000
            explosion.Parent = game.Workspace
            newBomb:Destroy() -- Remove the bomb after explosion
        end
    end)

    debris:AddItem(newBomb, 10) -- Clean up bombs after 10 seconds
end

while true do
    spawnBomb()
    wait(1) -- Adjust the wait time for how often bombs fall
end
