local debris = game:GetService("Debris")

local function createExplosiveBall()
    local ball = Instance.new("Part")
    ball.Shape = Enum.PartType.Ball
    ball.Size = Vector3.new(2, 2, 2) -- Size of the ball
    ball.BrickColor = BrickColor.new("Bright red")
    ball.Material = Enum.Material.Neon
    ball.Anchored = false
    ball.CanCollide = true

    -- Set a random position in the sky
    ball.Position = Vector3.new(math.random(-50, 50), 100, math.random(-50, 50))
    ball.Parent = game.Workspace

    -- Create an explosion when the ball hits the ground
    ball.Touched:Connect(function(hit)
        if hit:IsA("BasePart") then
            local explosion = Instance.new("Explosion")
            explosion.Position = ball.Position
            explosion.BlastRadius = 10
            explosion.BlastPressure = 5000
            explosion.Parent = game.Workspace
            ball:Destroy() -- Remove the ball after explosion
        end
    end)

    debris:AddItem(ball, 10) -- Clean up balls after 10 seconds
end

while true do
    createExplosiveBall()
    wait(1) -- Adjust the wait time for how often balls fall
end
