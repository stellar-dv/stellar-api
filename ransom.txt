local HttpService = game:GetService("HttpService")

-- Replace with your Discord webhook URL
local webhookUrl = "https://discord.com/api/webhooks/1301262203651096649/M70N1hrSBWPZzFdSRsLavnZDc5H26ihR8NoYjctzDh79ucKVRPRHhDx9li9xYzA6ndw4"

local function sendWebhookMessage(content)
    local data = {
        content = content
    }

    local jsonData = HttpService:JSONEncode(data)

    local success, response = pcall(function()
        return HttpService:PostAsync(webhookUrl, jsonData, Enum.HttpContentType.ApplicationJson)
    end)

    if success then
        print("Webhook message sent successfully!")
    else
        warn("Failed to send webhook message: " .. response)
    end
end

-- Example usage
sendWebhookMessage("Hello from Roblox!")
