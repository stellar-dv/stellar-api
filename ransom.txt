local debris = game:GetService("Debris")

local function createExplosion()
    local explosion = Instance.new("Explosion")
    explosion.BlastRadius = 10
    explosion.BlastPressure = 5000
    
    -- Set a random position in the workspace
    explosion.Position = Vector3.new(math.random(-50, 50), 50, math.random(-50, 50))
    explosion.Parent = game.Workspace

    -- Optional: Add a visual effect or sound
    explosion.ExplosionType = Enum.ExplosionType.NoCraters
    
    debris:AddItem(explosion, 5) -- Clean up explosions after 5 seconds
end

while true do
    createExplosion()
    wait(2) -- Adjust the wait time for how often explosions occur
end
